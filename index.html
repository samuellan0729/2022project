#判斷BMI落在什麼區間上
df9['BMIRANGE'] = 0
for idx, row in df9.iterrows():  
    BMI = row['BMI']
    if BMI < 18.5:
        BMI_RANGE = '體重過輕'
    elif 18.5 <= BMI < 24.0:
        BMI_RANGE = '正常範圍'
    elif 24.0 <= BMI < 27.0:
        BMI_RANGE = '體重過重'
    elif 27.0 <= BMI < 30.0: 
        BMI_RANGE = '輕度肥胖'
    elif 30.0 <= BMI < 35.0:
        BMI_RANGE = '中度肥胖'
    else:
        BMI_RANGE = '重度肥胖'
    df9.loc[idx, 'BMIRANGE'] = BMI_RANGE
    df9.loc[idx, 'BMIRANGE2'] = 1
